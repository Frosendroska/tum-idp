% Make chapters continuous (don't start on new pages)  
% Completely override chapter command to prevent page breaks
\makeatletter
\renewcommand*{\@makechapterhead}[1]{%
  \vspace{1\baselineskip}%
  {\parindent \z@ \raggedright \normalfont
    \ifnum \c@secnumdepth >\m@ne
      \if@mainmatter
        \Large\bfseries \thechapter.\space #1\par\nobreak
      \else
        \Large\bfseries #1\par\nobreak  
      \fi
    \else
      \Large\bfseries #1\par\nobreak
    \fi
    \vskip 0.5\baselineskip
  }%
}

\renewcommand*{\@makeschapterhead}[1]{%
  \vspace{1\baselineskip}%
  {\parindent \z@ \raggedright
    \normalfont
    \interlinepenalty\@M
    \Large \bfseries #1\par\nobreak
    \vskip 0.5\baselineskip
  }%
}

% Override the chapter command completely to remove all page breaking
\renewcommand\chapter{%
  \@ifstar{\@schapter}{\@dblarg\@chapter}%
}

\def\@chapter[#1]#2{%
  \ifnum \c@secnumdepth >\m@ne
    \if@mainmatter
      \refstepcounter{chapter}%
      \typeout{\thechapter.\space#1}%
      \addcontentsline{toc}{chapter}%
                       {\protect\numberline{\thechapter}#1}%
    \else
      \addcontentsline{toc}{chapter}{#1}%
    \fi
  \else
    \addcontentsline{toc}{chapter}{#1}%
  \fi
  \chaptermark{#1}%
  \@makechapterhead{#2}%
}

\def\@schapter#1{%
  \@makeschapterhead{#1}%
  \chaptermark{#1}%
  \addcontentsline{toc}{chapter}{#1}%
}
\makeatother
